+# Analytics reporting helpers
+
+def double_items(lst):
+    """Return a new list with each element doubled."""
+    result = []
+    for i in range(len(lst)):   # STYLE: prefer direct iteration or a list comprehension
+        result.append(lst[i] * 2)
+    return result
+
+def mean(numbers):
+    """Return the arithmetic mean, or 0.0 for an empty list."""
+    if not numbers:
+        return 0.0
+    total = 0
+    for n in numbers:
+        total = total + n       # STYLE: use sum(numbers) instead of manual accumulation
+    return total / len(numbers)
+
+def merge_configs(base, overrides):
+    """Merge two config dicts; values in overrides take precedence."""
+    merged = {}
+    for k in base:
+        merged[k] = base[k]
+    for k in overrides:
+        merged[k] = overrides[k]  # STYLE: use {**base, **overrides} or base | overrides (3.9+)
+    return merged
+
+class ReportBuilder:
+    def __init__(self):
+        self.rows = []
+        self._is_built = False  # STYLE: bool flag is redundant — len(self.rows) > 0 suffices
+
+    def add_row(self, row):
+        self.rows.append(row)
+
+    def build(self):
+        self.rows = double_items(self.rows)
+        self._is_built = True
+        return self.rows
