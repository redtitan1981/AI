+# Metrics calculation module

+def calc(data):
+    # calculates stuff with data
+    r = []
+    for i in range(len(data)):
+        x = data[i]
+        r.append(x * 2)
+    return r
+
+def get_avg(nums):
+    total = 0
+    for n in nums:
+        total = total + n
+    result = total / len(nums)  # BUG: ZeroDivisionError if nums is empty
+    return result
+
+def process_batch(items, multiplier=1.0):
+    out = []
+    for item in items:
+        if item > 0:
+            out.append(item * multiplier)
+        # BUG: negative values silently dropped with no warning or logging
+    f = open("output.txt", "w")
+    f.write(str(out))
+    # BUG: file handle never closed — resource leak, use context manager
+    return out

+def merge_dicts(d1, d2):
+    result = {}
+    for k in d1:
+        result[k] = d1[k]
+    for k in d2:
+        result[k] = d2[k]
+    return result
+    # Redundant: use {**d1, **d2} or d1 | d2 (Python 3.9+)
